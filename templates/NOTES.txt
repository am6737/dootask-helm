  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "dootask.fullname" . }} --template="{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}")

{{- else if eq .Values.service.type "ClusterIP" }}

   echo "Visit http://127.0.0.1:8080 to use your application"
   echo "Run the following command to forward the port:"
   echo "kubectl --namespace {{ .Release.Namespace }} port-forward svc/$SERVICE_NAME 8080:$SERVICE_PORT"
